window.addEvent('domready',function(){
	$$('.scrollbars.osx').scrollbars({
		scrollBarSize:10,
		barOverContent:true,
		fade:true // default 500 milliseconds
	});
});
/* Set inactive scrollbars to display none */
.scrollbars.osx ul.scrollbar.inactive { display: none; }

/* No scrollbar buttons */
.scrollbars.osx ul.scrollbar li.scroll.down,
.scrollbars.osx ul.scrollbar li.scroll.up,
.scrollbars.osx ul.scrollbar li.scroll.left,
.scrollbars.osx ul.scrollbar li.scroll.right{
	display: none;
}

/* Move the bars a bit from the side and make them a little bit smaller then its wrapper*/
.scrollbars.osx ul.scrollbar.vertical li.scroll.bar-wrapper{
	min-width: 7px !important;
	right: 2px !important; }
.scrollbars.osx ul.scrollbar.horizontal li.scroll.bar-wrapper{
	min-height: 7px !important;
	bottom: 2px !important;
}

/* Scrollbar style */
.scrollbars.osx ul.scrollbar li.scroll.bar-wrapper div.scroll.bar{
	background: black;
	opacity: .5;
	border-radius: 5px;
}
<div class="tab-pane row active" id="example-osx">
	<div class="span3 scrollbars osx" style="height: 90px">
		<h4>Scroll vertically</h4>
		...
	</div>
	<div class="span3 scrollbars osx" style="height: 90px">
		<h4>Scroll both ways</h4>
		<p style="width:400px; margin:0;">
			...
		</p>
	</div>
	<div class="span3 scrollbars osx">
		<h4>No scroll needed (inactive)</h4>
		...
	</div>
</div>

window.addEvent('domready',function(){
	$$('.scrollbars.invert').scrollbars({
		scrollBarSize:10
	});
});
/* Set inactive scrollbars to display none, but keep the wrapper displayed */
.scrollbars.invert ul.scrollbar.inactive li.scroll.bar-wrapper{ display: none; }

/* No scrollbar buttons */
.scrollbars.invert ul.scrollbar li.scroll.down,
.scrollbars.invert ul.scrollbar li.scroll.up,
.scrollbars.invert ul.scrollbar li.scroll.left,
.scrollbars.invert ul.scrollbar li.scroll.right{
	display: none;
}

/* Move the bars a bit from the side and make them a little bit smaller then its wrapper*/
.scrollbars.invert ul.scrollbar.vertical li.scroll.bar-wrapper{
	min-width: 7px !important;
	right: 2px !important; }
.scrollbars.invert ul.scrollbar.horizontal li.scroll.bar-wrapper{
	min-height: 7px !important;
	bottom: 2px !important;
}

/* Scrollbar style */
.scrollbars.invert ul.scrollbar { background: #000; }
/* Fill up the space between the vertical and horizontal bars */
.scrollbars.invert ul.scrollbar.horizontal:after {
	content: "";
	display: block;
	height: 100%;
	width: 100%;
	line-height: 0;
	background: black;
	margin-right: -10px;
	float: right;
}
.scrollbars.invert ul.scrollbar li.scroll.bar-wrapper div.scroll.bar{
	background: white;
	border-radius: 5px;
}
